\begin{column}{\colwidth}

\begin{alertblock}{The Problem in One Sentence}
Given only a finished song, recover its hidden tracks, the \leadA{vocals} and the
\leadA{accompaniment}, that were summed together when it was mixed.
\vskip0.4ex
This is hard because mixing destroys information: countless track pairs add up to
the exact same song. We need a \textbf{prior} that says which split is plausible.
\end{alertblock}

\begin{block}{Intuition: a song is a picture of sound}
\begin{itemize}
\item \lead{Spectrogram.} The short-time Fourier transform turns audio into an
image: time on the $x$-axis, pitch on the $y$-axis, brightness = energy.
\item \lead{Vocals are sparse.} A singing voice lights up only a few horizontal
harmonic lines per instant, most of the image is dark.
\item \lead{Accompaniment is low-rank.} Repeating chords and drums make the
background a sum of a few recurring patterns.
\end{itemize}
\vskip0.3ex
\textbf{Sparse} and \textbf{low-rank} are exactly the structures that
\textbf{compressed sensing} knows how to recover, so the same math that
reconstructs MRI and photographs from few measurements applies to audio.
\end{block}

\begin{block}{The structure each method exploits}
\begin{figure}
\includegraphics[width=\linewidth]{fig_triptych.pdf}
\caption{Mixture (left), the vocal track (center, sparse harmonic scaffold), and the
accompaniment (right, denser and low-rank). Our methods reverse the sum.}
\end{figure}
\end{block}

\begin{block}{When is recovery guaranteed?}
For $\bm{y}=\bm{A}\bm{x}$ with $\bm{x}$ $s$-sparse, the convex program
$\min\norm{\bm{z}}_1$ s.t.\ $\bm{A}\bm{z}=\bm{y}$ returns $\bm{x}$ exactly under
\emph{any} of three conditions:
\[ \underbrace{\spark(\bm{A})>2s}_{\text{uniqueness}},\quad
   \underbrace{\delta_{2s}<\sqrt2-1}_{\text{RIP}},\quad
   \underbrace{s<\tfrac12\!\big(1+\tfrac1{\mu(\bm{A})}\big)}_{\text{coherence}}. \]
\begin{itemize}
\item \lead{RIP:} $(1-\delta_s)\norm{\bm{x}}_2^2\le\norm{\bm{A}\bm{x}}_2^2\le(1+\delta_s)\norm{\bm{x}}_2^2$.
\item \lead{Coherence:} $\mu(\bm{A})=\max_{i\ne j}|\langle\bm{a}_i,\bm{a}_j\rangle|$; random
$\bm{A}$ needs only $m\!\ge\!Cs\log(N/s)$ rows.
\item \lead{Low-rank:} nuclear norm $\norm{\bm{X}}_*\!=\!\sum_i\sigma_i$ is the convex
surrogate for rank.
\end{itemize}
\end{block}

\begin{block}{Classical recovery stages}
\lead{Robust PCA.} Split the spectrogram into low-rank + sparse:
\[ \min_{\bm{X},\bm{E}}\norm{\bm{X}}_*+\lambda\norm{\bm{E}}_1
\ \text{s.t.}\ \bm{X}\!+\!\bm{E}\!=\!\bm{M},\ \lambda\!=\!\tfrac1{\sqrt{\max(F,T)}}. \]
Low-rank $\bm{X}$ = accompaniment, sparse $\bm{E}$ = voice. No training.
\vskip0.4ex
\lead{Supervised NMF.} Learn a non-negative dictionary per source; KL
multiplicative updates
$\bm{H}\!\leftarrow\!\bm{H}\!\odot\!\frac{\bm{D}^\top(\bm{M}\oslash\bm{D}\bm{H})}{\bm{D}^\top\bm1}$,
fix $\bm{D}\!=\![\bm{D}_v|\bm{D}_a]$ at test time.
\vskip0.4ex
\lead{K-SVD + SRC.} Overcomplete dictionaries via rank-one atom updates; route each
frame by which source reconstructs it best,
$c^\star\!=\!\argmin_c\norm{\bm{y}-\bm{D}_c\bm{\alpha}_c}_2$.
\end{block}

\begin{alertblock}{Deep sparse coder (the new method)}
\leadA{Idea.} A deep network \emph{is} a sparse-recovery solver with its steps
unrolled into layers. We learn the dictionaries by backprop. Each module stacks a
\emph{fat} (upsampling) then \emph{tall} (downsampling) non-negative code:
\[ \bm{\alpha}^{(1)}\!=\!\argmin_{\bm{\alpha}\ge0}\tfrac12\norm{\bm{x}\!-\!\bm{D}^{(1)}\bm{\alpha}}_2^2
+\lambda_1\norm{\bm{\alpha}}_1+\tfrac{\lambda_2}2\norm{\bm{\alpha}}_2^2, \]
solved by $K$ unrolled proximal-gradient steps (all matmul + clamp, runs on GPU):
\[ \bm{\alpha}_k=\big[\bm{z}_k-\tfrac1L\bm{D}^\top(\bm{D}\bm{z}_k-\bm{x})-\tfrac{\lambda_1}L\big]_+. \]
\leadA{Training.} Stack 3 modules + sigmoid; minimize
$\norm{(\bm{M}_v-\bm{M}_v^{\mathrm{IRM}})\!\odot\!\bm{M}}_1+\beta\sum_\ell\norm{\bm{\alpha}^{(\ell)}}_1$.
This is LISTA, specialized to non-negative codes and a masking output.
\end{alertblock}

\begin{block}{Stereo as joint sparsity}
\lead{Two channels, one support.} Left and right share the same active sources, so
their codes share \emph{rows}. Simultaneous OMP scores atoms by the cross-channel
row norm $\norm{(\bm{D}^\top\bm{R})_{n,:}}_2$, and the shared support \emph{improves}
identifiability: $s<\tfrac12(\spark(\bm{A})-1+\operatorname{rank}(\bm{Y}))$.
\end{block}

\begin{block}{Honest negative result: coherence}
\begin{figure}
\includegraphics[width=0.55\linewidth]{fig_ksvd_coherence.pdf}
\end{figure}
Learned dictionaries are \textbf{coherent} ($\mu\!\approx\!0.85$), so the worst-case
theorem certifies recovery only for $s\!\lesssim\!1$, yet $s\!=\!10$ works in
practice. \lead{Average-case $\ne$ worst-case} for trained dictionaries.
\end{block}

\begin{alertblock}{Takeaways}
\begin{itemize}
\item \leadA{Unification.} One front end places low-rank, NMF, dictionary, and deep
methods on a single structure-recovery axis.
\item \leadA{Deep wins.} The unrolled sparse coder is the most accurate non-oracle
separator \emph{and} the fastest learned model.
\item \leadA{More structure-learning $\Rightarrow$ better separation}, monotonically.
\item \leadA{Code, models, audio:} \texttt{github.com/takakhoo/ENGS109\_Final\_Project}
\end{itemize}
\end{alertblock}
