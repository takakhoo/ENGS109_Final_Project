\begin{alertblock}{Problem \& Contribution}
\textbf{Music source separation} recovers stems (vocals, accompaniment) from a
single mixture. Classical decompositions and large neural networks are usually
treated as separate worlds. \textbf{We show they are points on one axis:
recovery of structured (sparse / low-rank) spectrogram representations.}
\begin{itemize}
\item Four separators on a shared STFT front end: low-rank+sparse, supervised
NMF, discriminative K-SVD, and a \textbf{deep unrolled sparse coder}.
\item The deep sparse coder is the \textbf{strongest non-oracle separator and
the fastest learned model}.
\item Honest \textbf{coherence analysis}: where the recovery theory binds and
where it goes slack.
\end{itemize}
\end{alertblock}

\begin{block}{The structured-recovery view}
On the magnitude STFT $\bm{M}=|\STFT(y)|$: vocals are \textbf{sparse} on a
formant scaffold; accompaniment is \textbf{low-rank} (sustained harmony).
\begin{figure}
\includegraphics[width=\linewidth]{fig_triptych.pdf}
\caption{Mixture vs.\ vocals vs.\ accompaniment. The structure each method exploits.}
\end{figure}
\end{block}

\begin{block}{Preliminaries: when does recovery hold?}
For $\bm{y}=\bm{A}\bm{x}$, $\bm{x}$ $s$-sparse, the convex program
$\min\norm{\bm{z}}_1$ s.t. $\bm{A}\bm{z}=\bm{y}$ recovers $\bm{x}$ under any of:
\[ \spark(\bm{A})>2s, \qquad \delta_{2s}<\sqrt2-1, \qquad
s<\tfrac12\!\left(1+\tfrac1{\mu(\bm{A})}\right). \]
RIP: $(1-\delta_s)\norm{\bm{x}}_2^2\le\norm{\bm{A}\bm{x}}_2^2\le(1+\delta_s)\norm{\bm{x}}_2^2$.
Coherence $\mu(\bm{A})=\max_{i\ne j}|\langle\bm{a}_i,\bm{a}_j\rangle|$.
Random $\bm{A}$ needs $m\ge C\,s\log(N/s)$ measurements.
\vskip0.5ex
\textbf{Low-rank:} nuclear norm $\norm{\bm{X}}_*=\sum_i\sigma_i$ is the convex
surrogate for rank; principal component pursuit splits $\bm{M}=\bm{X}+\bm{E}$.
\end{block}

\begin{block}{Joint sparsity for stereo (MMV)}
Stereo channels share active sources, so their codes share a \textbf{row support}.
Stack per-frame channel spectra into $\bm{Y}=[\bm{y}_L\ \bm{y}_R]$ and run
\textbf{simultaneous OMP}, scoring atoms by the row norm
$\norm{(\bm{D}^\top\bm{R})_{n,:}}_2$ so both channels agree on active atoms.
Mixed-norm surrogate $\norm{\bm{X}}_{1,2}=\sum_n\norm{\bm{X}_{n,:}}_2$.
Shared support \emph{improves} identifiability:
\[ s<\tfrac12\big(\spark(\bm{A})-1+\mathrm{rank}(\bm{Y})\big). \]
\end{block}

\begin{block}{Classical recovery stages}
\textbf{Low-rank + sparse (RPCA).} Principal component pursuit,
\[ \min_{\bm{X},\bm{E}}\ \norm{\bm{X}}_*+\lambda\norm{\bm{E}}_1
\ \text{s.t.}\ \bm{X}+\bm{E}=\bm{M},\quad \lambda=\tfrac1{\sqrt{\max(F,T)}}, \]
solved by inexact ALM ($\SVT$ + soft-threshold). Low-rank $\bm{X}$=accompaniment.
\vskip0.4ex
\textbf{Supervised NMF.} Per-source non-negative dictionaries, KL multiplicative
updates $\bm{H}\!\leftarrow\!\bm{H}\odot\frac{\bm{D}^\top(\bm{M}\oslash\bm{D}\bm{H})}{\bm{D}^\top\bm1}$;
fix $\bm{D}=[\bm{D}_v|\bm{D}_a]$ at test time.
\vskip0.4ex
\textbf{K-SVD + SRC.} Overcomplete dictionaries via rank-one atom updates; route
each frame by $c^\star=\argmin_c\norm{\bm{y}-\bm{D}_c\bm{\alpha}_c}_2$.
\end{block}

\begin{alertblock}{Deep sparse coding network (novel)}
Predict the mask by \textbf{unrolling a non-negative proximal solver and learning
its dictionaries}. Each composite module: fat (upsampling) then tall
(downsampling) sparse code,
\[ \bm{\alpha}^{(1)}=\argmin_{\bm{\alpha}\ge0}\tfrac12\norm{\bm{x}-\bm{D}^{(1)}\bm{\alpha}}_2^2
+\lambda_1\norm{\bm{\alpha}}_1+\tfrac{\lambda_2}2\norm{\bm{\alpha}}_2^2, \]
\[ \bm{\alpha}_{k}=\Big[\bm{z}_k-\tfrac1L\bm{D}^\top(\bm{D}\bm{z}_k-\bm{x})
-\tfrac{\lambda_1}L\Big]_+\ \text{(unrolled FISTA)}. \]
Stack 3 modules + sigmoid head; train end-to-end on
$\norm{(\bm{M}_v-\bm{M}_v^{\mathrm{IRM}})\odot\bm{M}}_1+\beta\sum_\ell\norm{\bm{\alpha}^{(\ell)}}_1$.
This is LISTA specialized to non-negative codes and a masking output.
\end{alertblock}

\begin{block}{Coherence: an honest negative result}
\begin{figure}
\includegraphics[width=0.72\linewidth]{fig_ksvd_coherence.pdf}
\end{figure}
Learned dictionaries are coherent ($\mu\approx0.85$), so the worst-case bound
certifies exact recovery only for $s\lesssim1$, yet OMP at $s{=}10$ separates well.
\textbf{Average-case $\ne$ worst-case} for trained dictionaries.
\end{block}

\begin{alertblock}{Takeaways}
\begin{itemize}
\item One STFT front end unifies low-rank, NMF, dictionary, and deep methods.
\item \textbf{Unrolled deep sparse coding}: best non-oracle, fastest learned model.
\item Increasing structure-learning $\Rightarrow$ monotone quality gain.
\item Code + models + audio: \texttt{github.com/takakhoo/ENGS109\_Final\_Project}
\end{itemize}
\end{alertblock}
